import com.sun.net.httpserver.HttpServer; import com.sun.net.httpserver.HttpHandler; import com.sun.net.httpserver.HttpExchange; import java.io.*; import java.net.InetSocketAddress; import java.net.URI; import java.util.*; public class RefereeAnalyzerServer { static List decisions = new ArrayList<>(); public static void main(String[] args) throws Exception { loadDecisions(); HttpServer server = HttpServer.create(new InetSocketAddress(8080),0); server.createContext("/", new HomeHandler()); server.createContext("/analyze", new AnalyzeHandler()); server.setExecutor(null); System.out.println("Referee Analyzer Server running on http://localhost:8080"); server.start(); } static void loadDecisions() { decisions.add(new Decision("Foul im Strafraum", "Elfmeter geben")); decisions.add(new Decision("Abseits Tor", "Tor zurücknehmen")); decisions.add(new Decision("Handspiel", "Freistoß")); decisions.add(new Decision("Notbremse", "Rot Karte")); decisions.add(new Decision("Fehlalarm", "Spiel weiterlaufen lassen")); } static class Decision { String situation; String correctAction; public Decision(String situation, String correctAction){ this.situation = situation; this.correctAction = correctAction; } } static class HomeHandler implements HttpHandler { @Override public void handle(HttpExchange exchange) throws IOException { String page = """ Schiedsrichter Analyse

Schiedsrichter Fehlentscheidungen Analyse



"""; send(exchange, page); } } static class AnalyzeHandler implements HttpHandler { @Override public void handle(HttpExchange exchange) throws IOException { String body = new String(exchange.getRequestBody().readAllBytes()); Map params = parseQuery(body); String situation = params.getOrDefault("situation","Unbekannt"); String result = analyzeDecision(situation); String page = """ Analyse Ergebnis

Analyse Ergebnis

Situation: """ + situation + "

" + "

Empfohlene Entscheidung: " + result + "

" + "
Zurück" + "
" + ""; send(exchange,page); } } static String analyzeDecision(String situation){ for(Decision d : decisions){ if(d.situation.equalsIgnoreCase(situation)){ return d.correctAction; } } return "Keine Empfehlung verfügbar"; } static void send(HttpExchange exchange,String response) throws IOException { exchange.sendResponseHeaders(200,response.getBytes().length); OutputStream os = exchange.getResponseBody(); os.write(response.getBytes()); os.close(); } static Map parseQuery(String query){ Map map = new HashMap<>(); if(query==null) return map; String[] pairs = query.split("&"); for(String p:pairs){ String[] keyValue = p.split("="); if(keyValue.length==2){ map.put(keyValue[0],keyValue[1].replace("+"," ")); } } return map; } }
